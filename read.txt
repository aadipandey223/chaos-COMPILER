import { execSync } from 'child_process';
import { createHash } from 'crypto';

interface LingoConfig {
  apiKey: string;
  projectId: string;
  baseLocale: 'en';
  targetLocales: ['es', 'zh', 'hi', 'fr', 'de', 'ja']; // Top CS education languages
}

interface TranslationPayload {
  key: string;
  source: string;
  context: string; // Compiler context: "explanation", "ui", "error"
  glossaryTerms: string[]; // Terms that must not be translated literally
}

export class LingoCLI {
  private config: LingoConfig;
  
  constructor(config: LingoConfig) {
    this.config = config;
  }

  /**
   * Push source content to Lingo.dev for translation
   * Preserves compiler terminology via glossary protection
   */
  async pushSource(translations: TranslationPayload[]): Promise<void> {
    // Validate glossary terms are marked correctly
    const validatedSources = translations.map(t => ({
      ...t,
      // Ensure terms like "IR", "AST", "Opaque Predicate" have glossary protection
      source: this.wrapGlossaryTerms(t.source, t.glossaryTerms)
    }));

    // Execute CLI push
    try {
      execSync(`npx lingo.dev push \
        --project=${this.config.projectId} \
        --api-key=${this.config.apiKey} \
        --format=json`, 
        { input: JSON.stringify(validatedSources), encoding: 'utf-8' }
      );
    } catch (error) {
      throw new Error(`Lingo push failed: ${error.message}`);
    }
  }

  /**
   * Pull translations with validation
   * Critical: Validate that compiler terms maintain semantic accuracy
   */
  async pullTranslations(locale: string): Promise<Record<string, string>> {
    const raw = execSync(`npx lingo.dev pull \
      --project=${this.config.projectId} \
      --locale=${locale} \
      --api-key=${this.config.apiKey} \
      --format=json`
    ).toString();

    const translations = JSON.parse(raw);
    
    // Post-pull validation: Ensure compiler terminology integrity
    return this.validateCompilerTerminology(translations, locale);
  }

  /**
   * Critical for compiler education: 
   * Verify that technical terms maintain pedagogical accuracy
   */
  private validateCompilerTerminology(
    translations: Record<string, string>, 
    locale: string
  ): Record<string, string> {
    const validations = {
      // Terms that must preserve specific technical meaning
      'ir': ['中间表示', 'representación intermedia', 'représentation intermédiaire'],
      'ast': ['抽象语法树', 'árbol de sintaxis abstracta', 'arbre de syntaxe abstraite'],
      'opaque_predicate': ['不透明谓词', 'predicado opaco', 'prédicat opaque'],
      'ssa': ['静态单赋值', 'asignación estática simple', 'assignation statique unique']
    };

    // Check if validated translations match approved glossary
    Object.entries(validations).forEach(([term, approvedTranslations]) => {
      const translated = translations[`glossary.${term}`];
      if (translated && !approvedTranslations.includes(translated)) {
        console.warn(
          `⚠️  Terminology drift detected: "${term}" → "${translated}"\n` +
          `Approved: ${approvedTranslations.join(', ')}\n` +
          `Context: Compiler concepts require precise translation.`
        );
      }
    });

    return translations;
  }

  private wrapGlossaryTerms(text: string, terms: string[]): string {
    // Wrap glossary terms in Lingo.dev protected tags
    // Prevents translators from altering technical terms
    let protectedText = text;
    terms.forEach(term => {
      const regex = new RegExp(`\\b${term}\\b`, 'gi');
      protectedText = protectedText.replace(regex, `<glossary>${term}</glossary>`);
    });
    return protectedText;
  }
}

interface MultilingualExplanation {
  en: string;
  localized: Record<string, string>;
  validationStatus: Record<string, 'verified' | 'pending' | 'rejected'>;
}

export class MultilingualMCPService {
  private lingoCLI: LingoCLI;
  private currentLocale: string = 'en';
  
  /**
   * Generate explanation with automatic translation validation
   */
  async generateLocalizedExplanation(
    diagnostic: Diagnostic,
    targetLocale: string
  ): Promise<MultilingualExplanation> {
    // 1. Generate English explanation (source of truth)
    const enExplanation = await this.generateEnglishExplanation(diagnostic);
    
    // 2. If English, return immediately
    if (targetLocale === 'en') {
      return {
        en: enExplanation,
        localized: { en: enExplanation },
        validationStatus: { en: 'verified' }
      };
    }

    // 3. Check translation memory via Lingo CLI
    const cached = await this.getCachedTranslation(diagnostic.id, targetLocale);
    
    if (cached && cached.validationStatus === 'verified') {
      return {
        en: enExplanation,
        localized: { [targetLocale]: cached.text },
        validationStatus: { [targetLocale]: 'verified' }
      };
    }

    // 4. Trigger translation workflow (async)
    // In production: Send to Lingo.dev for professional translation
    // For hackathon/demo: Use AI translation with glossary enforcement
    const translated = await this.translateWithGlossary(
      enExplanation, 
      targetLocale,
      this.extractGlossaryTerms(diagnostic)
    );

    // 5. Validate translation maintains compiler accuracy
    const isValid = await this.lingoCLI.validateTranslation(
      enExplanation,
      translated,
      targetLocale
    );

    return {
      en: enExplanation,
      localized: { [targetLocale]: translated },
      validationStatus: { 
        [targetLocale]: isValid ? 'verified' : 'rejected' 
      }
    };
  }

  /**
   * Critical: Ensure translated explanations use correct compiler terminology
   */
  private async translateWithGlossary(
    text: string, 
    locale: string,
    glossaryTerms: string[]
  ): Promise<string> {
    // Replace terms with placeholders before translation
    let processed = text;
    const placeholders: Record<string, string> = {};
    
    glossaryTerms.forEach((term, idx) => {
      const placeholder = `__TERM_${idx}__`;
      placeholders[placeholder] = term;
      processed = processed.replace(new RegExp(`\\b${term}\\b`, 'g'), placeholder);
    });

    // Translate (via AI or Lingo CLI)
    const translated = await this.machineTranslate(processed, locale);

    // Restore glossary terms
    let result = translated;
    Object.entries(placeholders).forEach(([placeholder, term]) => {
      // Get localized term from glossary
      const localizedTerm = this.getLocalizedTerm(term, locale);
      result = result.replace(new RegExp(placeholder, 'g'), localizedTerm);
    });

    return result;
  }

  private getLocalizedTerm(term: string, locale: string): string {
    // Your glossary.json with multilingual support
    const glossary = {
      'en': { 'ir': 'Intermediate Representation', 'ast': 'Abstract Syntax Tree' },
      'es': { 'ir': 'Representación Intermedia', 'ast': 'Árbol de Sintaxis Abstracta' },
      'zh': { 'ir': '中间表示', 'ast': '抽象语法树' },
      'hi': { 'ir': 'इंटरमीडिएट प्रतिनिधित्व', 'ast': 'अमूर्त सिंटैक्स ट्री' }
    };
    
    return glossary[locale]?.[term] || term;
  }
}

import React, { createContext, useContext, useState, useEffect } from 'react';
import { LingoCLI } from '@/lingo/cli-integration';

interface I18nContextType {
  locale: string;
  setLocale: (locale: string) => void;
  translations: Record<string, string>;
  validationStatus: 'loading' | 'verified' | 'partial' | 'error';
  availableLocales: string[];
}

const I18nContext = createContext<I18nContextType | undefined>(undefined);

export const LanguageProvider: React.FC = ({ children }) => {
  const [locale, setLocale] = useState('en');
  const [translations, setTranslations] = useState({});
  const [status, setStatus] = useState('loading');
  
  const lingo = new LingoCLI({
    apiKey: import.meta.env.VITE_LINGO_API_KEY,
    projectId: 'chaos-lab-compiler',
    baseLocale: 'en',
    targetLocales: ['es', 'zh', 'hi', 'fr']
  });

  useEffect(() => {
    if (locale === 'en') {
      setTranslations({});
      setStatus('verified');
      return;
    }

    // Fetch validated translations from Lingo
    lingo.pullTranslations(locale)
      .then(data => {
        setTranslations(data);
        setStatus('verified');
      })
      .catch(err => {
        console.error('Translation fetch failed:', err);
        setStatus('error');
        // Fallback to English with warning
        setLocale('en');
      });
  }, [locale]);

  return (
    <I18nContext.Provider value={{
      locale,
      setLocale,
      translations,
      validationStatus: status as any,
      availableLocales: ['en', 'es', 'zh', 'hi', 'fr']
    }}>
      {children}
    </I18nContext.Provider>
  );
};

export const useI18n = () => {
  const ctx = useContext(I18nContext);
  if (!ctx) throw new Error('useI18n must be used within LanguageProvider');
  return ctx;
};

import { Plugin } from 'vite';
import { LingoCLI } from './src/lingo/cli-integration';

export function lingoPlugin(options: { 
  projectId: string; 
  pullOnBuild: boolean;
  validateGlossary: boolean;
}): Plugin {
  return {
    name: 'lingo-i18n',
    async buildStart() {
      if (!options.pullOnBuild) return;
      
      console.log('🌍 Fetching validated translations from Lingo.dev...');
      
      const lingo = new LingoCLI({
        apiKey: process.env.LINGO_API_KEY!,
        projectId: options.projectId,
        baseLocale: 'en',
        targetLocales: ['es', 'zh', 'hi', 'fr']
      });

      // Pull all translations
      const locales = ['es', 'zh', 'hi', 'fr'];
      for (const locale of locales) {
        try {
          const translations = await lingo.pullTranslations(locale);
          
          // Write to public folder for runtime loading
          const fs = await import('fs');
          fs.mkdirSync('./public/locales', { recursive: true });
          fs.writeFileSync(
            `./public/locales/${locale}.json`,
            JSON.stringify(translations, null, 2)
          );
          
          console.log(`✅ ${locale}: ${Object.keys(translations).length} strings validated`);
        } catch (error) {
          if (options.validateGlossary) {
            throw new Error(`Translation validation failed for ${locale}: ${error}`);
          }
          console.warn(`⚠️  Failed to fetch ${locale}, using fallbacks`);
        }
      }
    }
  };
}
